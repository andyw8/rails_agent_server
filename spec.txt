### Rails REPL

A persistent Rails REPL that avoids boot overhead for repeated `bin/rails runner` calls. Auto-starts a background server on first use.

```bash
bin/rails-repl 'User.count'                  # Run a Ruby expression (auto-starts server if needed)
bin/rails-repl /tmp/script.rb                # Run a script file
bin/rails-repl status                        # Check if server is running
bin/rails-repl stop                          # Stop the background server
bin/rails-repl restart                       # Restart the background server
```

---

#!/usr/bin/env ruby
# frozen_string_literal: true

# Persistent Rails REPL — avoids boot overhead for repeated queries.
# Auto-starts a background server on first use.
#
# Usage:
#   bin/rails-repl 'User.count'
#   bin/rails-repl path/to/script.rb
#   bin/rails-repl stop                 # stop the background server
#   bin/rails-repl restart              # restart the background server
#   bin/rails-repl status               # check if server is running

require "socket"
require "fileutils"

SOCKET_PATH = "/tmp/rails_repl.sock"
PID_PATH = "/tmp/rails_repl.pid"
LOG_PATH = "/tmp/rails_repl.log"

def run_server
  require_relative "../config/environment"

  FileUtils.rm_f(SOCKET_PATH)
  server = UNIXServer.new(SOCKET_PATH)
  File.write(PID_PATH, Process.pid.to_s)

  $stdout.puts "Rails REPL server listening on #{SOCKET_PATH} (PID: #{Process.pid})"

  at_exit do
    FileUtils.rm_f(SOCKET_PATH)
    FileUtils.rm_f(PID_PATH)
  end
  trap("INT") { exit }
  trap("TERM") { exit }

  loop do
    client = server.accept
    code = client.read

    output = StringIO.new
    result = nil
    error = nil

    begin
      old_stdout = $stdout
      $stdout = output
      result = eval(code) # rubocop:disable Security/Eval
      $stdout = old_stdout
    rescue => e
      $stdout = old_stdout
      error = "#{e.class}: #{e.message}"
    end

    printed = output.string
    response = +""
    response << printed unless printed.empty?
    if error
      response << error
    elsif printed.empty?
      response << result.inspect
    end

    client.write(response)
    client.close
  end
end

def server_running?
  return false unless File.exist?(PID_PATH)

  pid = File.read(PID_PATH).strip.to_i
  Process.kill(0, pid)
  true
rescue Errno::ESRCH, Errno::EPERM
  false
end

def stop_server
  if File.exist?(PID_PATH)
    pid = File.read(PID_PATH).strip.to_i
    Process.kill("TERM", pid)
    puts "Stopped Rails REPL server (PID: #{pid})"
  else
    puts "Rails REPL server is not running"
  end
rescue Errno::ESRCH
  puts "Rails REPL server is not running (stale PID file)"
  FileUtils.rm_f(PID_PATH)
end

def start_server
  puts "Starting Rails REPL server (this will take a few seconds)..."

  pid = spawn(
    RbConfig.ruby, __FILE__, "--server",
    out: LOG_PATH, err: LOG_PATH
  )
  Process.detach(pid)

  30.times do
    return if File.exist?(SOCKET_PATH)
    sleep 0.5
  end

  abort "Timed out waiting for Rails REPL server to start. Check #{LOG_PATH}"
end

def send_command(code)
  socket = UNIXSocket.new(SOCKET_PATH)
  socket.write(code)
  socket.close_write
  puts socket.read
  socket.close
end

# Internal: called by start_server to run the daemon in a subprocess
if ARGV[0] == "--server"
  run_server
  exit
end

case ARGV[0]
when "stop"
  stop_server
  exit
when "restart"
  stop_server if server_running?
  FileUtils.rm_f(SOCKET_PATH)
  sleep 0.5
  start_server
  puts "Rails REPL server restarted"
  exit
when "status"
  if server_running?
    pid = File.read(PID_PATH).strip
    puts "Rails REPL server is running (PID: #{pid})"
  else
    puts "Rails REPL server is not running"
  end
  exit
 end



code = if ARGV[0] && File.exist?(ARGV[0])
  File.read(ARGV[0])
else
  ARGV.join(" ")
end

abort "Usage: bin/rails-repl 'User.count' or bin/rails-repl script.rb" if code.empty?

start_server unless server_running?

send_command(code)
